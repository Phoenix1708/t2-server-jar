033    package uk.org.taverna.server.client;
034    
035    /**
036     * An enum to represent the status of a Run on a Taverna Server.
037     * 
038     * @author Robert Haines
039     */
040    public enum RunStatus {
041    
042            /**
043             * Used to represent a Run in the Initialized state.
044             */
045            INITIALIZED("Initialized"),
046    
047            /**
048             * Used to represent a Run in the Running state.
049             */
050            RUNNING("Operating"),
051    
052            /**
053             * Used to represent a Run in the Finished state.
054             */
055            FINISHED("Finished"),
056    
057            /**
058             * Used to represent a Run in the Stopped state. This state is currently
059             * unused.
060             */
061            STOPPED("Stopped"),
062    
063            /**
064             * Used to represent a Run that has been deleted from the server.
065             */
066            DELETED("Deleted"),
067    
068            /**
069             * Used to represent a Run in an undefined, or unknown state. This state is
070             * currently unused.
071             */
072            UNDEFINED("unknown");
073    
074            /**
075             * The String representation of this status.
076             */
077            public final String status;
078    
079            private RunStatus(String status) {
080                    this.status = status;
081            }
082    
083            /**
084             * Convert a String representation of a status to its RunStatus equivalent.
085             * 
086             * @param status
087             *            the status to convert.
088             * @return the converted status.
089             * @throws IllegalArgumentException
090             *             if there is no such status.
091             */
092            public static RunStatus fromString(String status) {
093                    for (RunStatus rs : RunStatus.values()) {
094                            if (rs.status.equals(status)) {
095                                    return rs;
096                            }
097                    }
098    
099                    throw new IllegalArgumentException("There is no such status as '"
100                                    + status + "'");
101            }
102    }
