033    package uk.org.taverna.server.client.xml;
034    
035    import java.io.ByteArrayOutputStream;
036    import java.io.File;
037    import java.net.URI;
038    
039    import javax.xml.bind.JAXBContext;
040    import javax.xml.bind.JAXBElement;
041    import javax.xml.bind.JAXBException;
042    import javax.xml.bind.Marshaller;
043    
044    import uk.org.taverna.server.client.RunPermission;
045    import uk.org.taverna.server.client.xml.rest.Credential;
046    import uk.org.taverna.server.client.xml.rest.InputDescription;
047    import uk.org.taverna.server.client.xml.rest.KeyPairCredential;
048    import uk.org.taverna.server.client.xml.rest.MakeDirectory;
049    import uk.org.taverna.server.client.xml.rest.ObjectFactory;
050    import uk.org.taverna.server.client.xml.rest.PasswordCredential;
051    import uk.org.taverna.server.client.xml.rest.Permission;
052    import uk.org.taverna.server.client.xml.rest.PermissionDescription;
053    import uk.org.taverna.server.client.xml.rest.TrustDescriptor;
054    
055    public final class XMLWriter {
056    
057            static byte[] write(JAXBElement<?> element) {
058                    ByteArrayOutputStream os = new ByteArrayOutputStream();
059                    try {
060                            JAXBContext context = JAXBContext
061                                            .newInstance("uk.org.taverna.server.client.xml.rest");
062                            Marshaller marshaller = context.createMarshaller();
063                            marshaller.marshal(element, os);
064                    } catch (JAXBException e) {
065                            // TODO Auto-generated catch block
066                            e.printStackTrace();
067                    }
068    
069                    return os.toByteArray();
070            }
071    
072            public static byte[] mkdir(String name) {
073                    MakeDirectory md = new MakeDirectory();
074                    md.setName(name);
075    
076                    ObjectFactory factory = new ObjectFactory();
077                    JAXBElement<MakeDirectory> element = factory.createMkdir(md);
078    
079                    return write(element);
080            }
081    
082            public static byte[] inputValue(String value) {
083                    InputDescription.Value idv = new InputDescription.Value();
084                    idv.setValue(value);
085    
086                    InputDescription id = new InputDescription();
087                    id.setValue(idv);
088    
089                    ObjectFactory factory = new ObjectFactory();
090                    JAXBElement<InputDescription> element = factory.createRunInput(id);
091    
092                    return write(element);
093            }
094    
095            public static byte[] inputFile(File file) {
096                    InputDescription.File idf = new InputDescription.File();
097                    idf.setValue(file.getPath());
098    
099                    InputDescription id = new InputDescription();
100                    id.setFile(idf);
101    
102                    ObjectFactory factory = new ObjectFactory();
103                    JAXBElement<InputDescription> element = factory.createRunInput(id);
104    
105                    return write(element);
106            }
107    
108            public static byte[] runPermission(String username, RunPermission permission) {
109                    PermissionDescription pd = new PermissionDescription();
110                    pd.setUserName(username);
111                    pd.setPermission(Permission.fromValue(permission.permission));
112    
113                    ObjectFactory factory = new ObjectFactory();
114                    JAXBElement<PermissionDescription> element = factory
115                                    .createPermissionUpdate(pd);
116    
117                    return write(element);
118            }
119    
120            public static byte[] runServiceUserPassCredential(URI uri, String username,
121                            String password) {
122                    PasswordCredential pc = new PasswordCredential();
123                    pc.setServiceURI(uri);
124                    pc.setUsername(username);
125                    pc.setPassword(password);
126    
127                    Credential cred = new Credential();
128                    cred.setUserpass(pc);
129    
130                    ObjectFactory factory = new ObjectFactory();
131                    JAXBElement<Credential> element = factory.createCredential(cred);
132    
133                    return write(element);
134            }
135    
136            public static byte[] runServiceKeyPairCredential(URI uri, String filename,
137                            String type, String name, String password) {
138                    KeyPairCredential kpc = new KeyPairCredential();
139                    kpc.setServiceURI(uri);
140                    kpc.setCredentialFile(filename);
141                    kpc.setCredentialName(name);
142                    kpc.setFileType(type);
143                    kpc.setUnlockPassword(password);
144    
145                    Credential cred = new Credential();
146                    cred.setKeypair(kpc);
147    
148                    ObjectFactory factory = new ObjectFactory();
149                    JAXBElement<Credential> element = factory.createCredential(cred);
150    
151                    return write(element);
152            }
153    
154            public static byte[] runTrustedIdentity(String filename, String type) {
155                    TrustDescriptor td = new TrustDescriptor();
156                    td.setCertificateFile(filename);
157                    td.setFileType(type);
158    
159                    ObjectFactory factory = new ObjectFactory();
160                    JAXBElement<TrustDescriptor> element = factory
161                                    .createTrustedIdentity(td);
162    
163                    return write(element);
164            }
165    }
