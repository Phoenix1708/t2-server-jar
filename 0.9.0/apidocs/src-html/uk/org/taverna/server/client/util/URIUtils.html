033    package uk.org.taverna.server.client.util;
034    
035    import java.net.URI;
036    import java.net.URISyntaxException;
037    
038    import org.apache.http.client.utils.URIBuilder;
039    
040    /**
041     * A small set of utility methods to ease working with URIs. URI objects are
042     * immutable which can be annoying to work with when string manipulation-like
043     * operations are required.
044     * 
045     * @author Robert Haines
046     * @since 0.9.0
047     */
048    public final class URIUtils {
049    
050            private URIUtils() {
051            }
052    
053            /**
054             * Strip any user info from a {@link URI}.
055             * 
056             * @param uri
057             *            the {@link URI} to strip.
058             * @return the stripped {@link URI}.
059             * @throws IllegalArgumentException
060             *             if a mal-formed {@link URI} is passed in.
061             */
062            public static URI stripUserInfo(final URI uri) {
063                    try {
064                            return new URIBuilder(uri).setUserInfo(null).build();
065                    } catch (URISyntaxException e) {
066                            throw new IllegalArgumentException("Bad URI passed in: " + uri);
067                    }
068            }
069    
070            /**
071             * Append a path to the end of an existing {@link URI}'s path.
072             * 
073             * @param uri
074             *            the {@link URI} on which to append the extra path.
075             * @param extraPath
076             *            the extra path to be appended to the {@link URI}.
077             * @return the new {@link URI} with the extended path.
078             * @throws IllegalArgumentException
079             *             if a mal-formed {@link URI} or path is passed in.
080             */
081            public static URI appendToPath(final URI uri, final String extraPath) {
082                    try {
083                            return new URIBuilder(uri).setPath(uri.getPath() + "/" + extraPath)
084                                            .build().normalize();
085                    } catch (URISyntaxException e) {
086                            throw new IllegalArgumentException(
087                                            "Either a bad URI was passed in (" + uri
088                                            + ") or an illegal path was appended to it ("
089                                            + extraPath + ")");
090                    }
091            }
092    
093            /**
094             * Extract the final component of the {@link URI}'s path.
095             * 
096             * If there is no path, or the path ends with <code>/</code> the empty
097             * {@link String} is returned. So:
098             * <ul>
099             * <li><code>http://example.org/this/is/a/path</code> would yield
100             * <code>path</code>,</li>
101             * <li><code>http://example.org</code> would yield the empty {@link String}
102             * and</li>
103             * <li><code>http://example.org/path/</code> would also yield the empty
104             * {@link String}.</li>
105             * </ul>
106             * 
107             * @param uri
108             *            the {@link URI} to get the final path component of.
109             * @return the final path component of the passed in {@link URI}.
110             */
111            public static String extractFinalPathComponent(final URI uri) {
112                    String path = uri.getPath();
113                    return path.substring(path.lastIndexOf("/") + 1);
114            }
115    
116            /**
117             * Set the query part of a {@link URI}.
118             * 
119             * @param uri
120             *            the {@link URI} to receive a new query part.
121             * @param query
122             *            the query to set (with or without the leading ?).
123             * @return the new {@link URI} with its query part set.
124             * @throws IllegalArgumentException
125             *             if a mal-formed {@link URI} or query is passed in.
126             */
127            public static URI setQuery(final URI uri, String query) {
128                    String q = query;
129                    if (q.startsWith("?")) {
130                            q = q.substring(1);
131                    }
132    
133                    try {
134                            return new URIBuilder(uri).setQuery(q).build();
135                    } catch (URISyntaxException e) {
136                            throw new IllegalArgumentException(
137                                            "Either a bad URI was passed in (" + uri
138                                            + ") or a bad query was appended to it (" + q + ")");
139                    }
140            }
141    }
