033    package uk.org.taverna.server.client;
034    
035    import java.net.URI;
036    
037    /**
038     * This exception is thrown if the server returns a forbidden error code (403).
039     * 
040     * @author Robert Haines
041     */
042    public class AccessForbiddenException extends ServerException {
043            private static final long serialVersionUID = 1L;
044    
045            private final URI uri;
046    
047            /**
048             * Construct the exception with the specified {@link URI}.
049             * 
050             * @param uri
051             *            the address that access was forbidden to.
052             */
053            public AccessForbiddenException(URI uri) {
054                    super(
055                                    "Access to "
056                                                    + uri
057                                                    + " is forbidden. The server does not allow the requested operation.");
058    
059                    this.uri = uri;
060            }
061    
062            /**
063             * Get the {@link URI} for which access was forbidden by the server.
064             * 
065             * @return the address that access was forbidden to.
066             */
067            public URI getURI() {
068                    return uri;
069            }
070    }
