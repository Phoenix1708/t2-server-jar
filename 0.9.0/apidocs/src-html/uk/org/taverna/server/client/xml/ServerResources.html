033    package uk.org.taverna.server.client.xml;
034    
035    import java.net.URI;
036    import java.util.Map;
037    
038    public final class ServerResources extends AbstractResources {
039    
040            private static final String SNAPSHOT = "-SNAPSHOT";
041    
042            private final String version;
043            private final String revision;
044            private final String timestamp;
045            private final int[] versionComponents;
046    
047            ServerResources(Map<ResourceLabel, URI> links, String version,
048                            String revision, String timestamp) {
049                    super(links);
050    
051                    this.revision = revision;
052                    this.timestamp = timestamp;
053    
054                    // Parse out and rebuild the version string into a normalized
055                    // major.minor.patch format. This is so that all version strings have
056                    // three components - 2.4 will be converted to 2.4.0.
057                    versionComponents = new int[3];
058                    if (version.endsWith(SNAPSHOT)) {
059                            version = version.replaceAll(SNAPSHOT, "");
060                    }
061                    String[] vs = version.split("\\.", 3);
062                    versionComponents[0] = Integer.parseInt(vs[0]);
063                    versionComponents[1] = Integer.parseInt(vs[1]);
064    
065                    try {
066                            // We might not have a third component!
067                            versionComponents[2] = Integer.parseInt(vs[2]);
068                    } catch (IndexOutOfBoundsException e) {
069                            versionComponents[2] = 0;
070                    }
071    
072                    this.version = String.format("%d.%d.%d", versionComponents[0],
073                                    versionComponents[1], versionComponents[2]);
074            }
075    
076            public String getVersion() {
077                    return version;
078            }
079    
080            public String getRevision() {
081                    return revision;
082            }
083    
084            public String getTimestamp() {
085                    return timestamp;
086            }
087    
088            public int[] getVersionComponents() {
089                    return versionComponents;
090            }
091    }
