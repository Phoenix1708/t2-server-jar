033    package uk.org.taverna.server.client;
034    
035    import java.io.File;
036    import java.io.IOException;
037    import java.io.InputStream;
038    import java.lang.ref.SoftReference;
039    import java.net.URI;
040    import java.util.Arrays;
041    
042    import org.apache.commons.lang.ArrayUtils;
043    import org.apache.commons.lang.math.LongRange;
044    
045    import uk.org.taverna.server.client.util.IOUtils;
046    
047    /**
048     * 
049     * @author Robert Haines
050     */
051    public class PortData extends PortValue {
052    
053            // If there is no data...
054            private static final byte[] EMPTY_DATA = new byte[0];
055    
056            // Data cache
057            private SoftReference<byte[]> cache;
058            private LongRange dataGot;
059    
060            PortData(Run run, URI reference, String type, long size) {
061                    super(run, reference, type, size);
062    
063                    cache = new SoftReference<byte[]>(null);
064                    dataGot = null;
065            }
066    
067            @Override
068            public boolean isError() {
069                    return false;
070            }
071    
072            @Override
073            public PortValue get(int index) throws IndexOutOfBoundsException {
074                    if (index != 0) {
075                            throw new IndexOutOfBoundsException();
076                    }
077    
078                    return this;
079            }
080    
081            @Override
082            public int size() {
083                    return 1;
084            }
085    
086            @Override
087            public long getDataSize() {
088                    return size;
089            }
090    
091            @Override
092            public InputStream getDataStream() {
093                    return run.getOutputDataStream(reference, null);
094            }
095    
096            @Override
097            public void writeDataToFile(File file) throws IOException {
098                    InputStream is = run.getOutputDataStream(reference, null);
099                    try {
100                            IOUtils.writeStreamToFile(is, file);
101                    } finally {
102                            org.apache.commons.io.IOUtils.closeQuietly(is);
103                    }
104            }
105    
106            @Override
107            public byte[] getData() {
108                    // LongRange is inclusive so size is too long by one.
109                    return getData(new LongRange(0, (size - 1)));
110            }
111    
112            @Override
113            public byte[] getData(int index) {
114                    return getData();
115            }
116    
117            public byte[] getData(int start, int length) {
118                    // If length is zero then there is nothing to return.
119                    if (length == 0) {
120                            return EMPTY_DATA;
121                    }
122    
123                    // LongRange is inclusive so (start + length) is too long by one.
124                    return getData(new LongRange(start, (start + length - 1)));
125            }
126    
127            private byte[] getData(LongRange range) {
128    
129                    byte[] data = cache.get();
130                    if (data == null) {
131                            dataGot = null;
132                    }
133    
134                    // Return empty data if this value is empty.
135                    if (getDataSize() == 0
136                                    || contentType.equalsIgnoreCase("application/x-empty")) {
137                            return EMPTY_DATA;
138                    }
139    
140                    // Check the range provided is sensible. LongRange is inclusive so size
141                    // is too long by one.
142                    if (range.getMinimumLong() < 0) {
143                            range = new LongRange(0, range.getMaximumLong());
144                    }
145                    if (range.getMaximumLong() >= getDataSize()) {
146                            range = new LongRange(range.getMinimumLong(),
147                                            (getDataSize() - 1));
148                    }
149    
150                    // Find the data range(s) that we need to download.
151                    LongRange[] need = fill(dataGot, range);
152    
153                    switch (need.length) {
154                    case 0:
155                            // We already have all the data we need, just return the right bit.
156                            // dataGot cannot be null here and must fully encompass range.
157                            int from = (int) (range.getMinimumLong() - dataGot.getMinimumLong());
158                            int to = (int) (range.getMaximumLong() - dataGot.getMinimumLong());
159    
160                            // copyOfRange is exclusive!
161                            return Arrays.copyOfRange(data, from, (to + 1));
162                    case 1:
163                            // we either have some data, at one end of range or either side of
164                            // it, or none. dataGot can be null here.
165                            // In both cases we download what we need.
166                            byte[] newData = run.getOutputData(reference, need[0]);
167                            if (dataGot == null) {
168                                    // This is the only data we have, return it all.
169                                    dataGot = range;
170                                    data = newData;
171                                    cache = new SoftReference<byte[]>(data);
172                                    return data;
173                            } else {
174                                    // Add the new data to the correct end of the data we have,
175                                    // then return the range requested.
176                                    if (range.getMaximumLong() <= dataGot.getMaximumLong()) {
177                                            dataGot = new LongRange(range.getMinimumLong(),
178                                                            dataGot.getMaximumLong());
179                                            data = ArrayUtils.addAll(newData, data);
180                                            cache = new SoftReference<byte[]>(data);
181    
182                                            // copyOfRange is exclusive!
183                                            return Arrays.copyOfRange(data, 0,
184                                                            (int) (range.getMaximumLong() + 1));
185                                    } else {
186                                            dataGot = new LongRange(dataGot.getMinimumLong(),
187                                                            range.getMaximumLong());
188                                            data = ArrayUtils.addAll(data, newData);
189                                            cache = new SoftReference<byte[]>(data);
190    
191                                            // copyOfRange is exclusive!
192                                            return Arrays.copyOfRange(data, (int) (range
193                                                            .getMinimumLong() - dataGot.getMinimumLong()),
194                                                            (int) (dataGot.getMaximumLong() + 1));
195                                    }
196                            }
197                    case 2:
198                            // We definitely have some data and it is in the middle of the
199                            // range requested. dataGot cannot be null here.
200                            dataGot = range;
201                            byte[] data1 = run.getOutputData(reference, need[0]);
202                            byte[] data2 = run.getOutputData(reference, need[1]);
203                            data = ArrayUtils.addAll(data1, data);
204                            data = ArrayUtils.addAll(data, data2);
205                            cache = new SoftReference<byte[]>(data);
206                            return data;
207                    }
208    
209                    // Should never get here! This is an error!
210                    return null;
211            }
212    
213            // Aaaarrrgh!
214            private LongRange[] fill(LongRange got, LongRange want) {
215                    if (got == null) {
216                            return new LongRange[] { want };
217                    }
218    
219                    if (got.containsLong(want.getMinimumLong())) {
220                            if (got.containsLong(want.getMaximumLong())) {
221                                    return new LongRange[0];
222                            } else {
223                                    return new LongRange[] { new LongRange(
224                                                    (got.getMaximumLong() + 1), want.getMaximumLong()) };
225                            }
226                    } else {
227                            if (got.containsLong(want.getMaximumLong())) {
228                                    return new LongRange[] { new LongRange(want.getMinimumLong(),
229                                                    (got.getMinimumLong() - 1)) };
230                            } else {
231                                    if (want.getMaximumLong() < got.getMinimumLong()) {
232                                            return new LongRange[] { new LongRange(
233                                                            want.getMinimumLong(), (got.getMinimumLong() - 1)) };
234                                    } else if (want.getMinimumLong() > got.getMaximumLong()) {
235                                            return new LongRange[] { new LongRange(
236                                                            (got.getMaximumLong() + 1), want.getMaximumLong()) };
237                                    } else {
238                                            return new LongRange[] {
239                                                            new LongRange(want.getMinimumLong(),
240                                                                            (got.getMinimumLong() - 1)),
241                                                                            new LongRange((got.getMaximumLong() + 1),
242                                                                                            want.getMaximumLong()) };
243                                    }
244                            }
245                    }
246            }
