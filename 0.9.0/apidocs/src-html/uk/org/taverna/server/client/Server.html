062    /**
063     * The Server class represents a connection to a Taverna Server instance
064     * somewhere on the Internet. Only one instance of this class is created for
065     * each Taverna Server instance.
066     * 
067     * To make a connection to a server call {@link Server#connect(URI)} with it
068     * full URL as the parameter. If there already exists a Server instance that is
069     * connected to this Taverna Server then it will be returned, otherwise a new
070     * Server instance is created and returned.
071     * 
072     * @author Robert Haines
073     */
074    public final class Server {
075    
076            /*
077             * Where to find the REST endpoint in relation to the base URI of a Taverna
078             * Server.
079             * 
080             * Add a slash to the end of this address to work around this bug:
081             * http://dev.mygrid.org.uk/issues/browse/TAVSERV-113
082             */
083            private final static String REST_ENDPOINT = "rest/";
084    
085            private final Connection connection;
086    
087            private final URI uri;
088            private final Map<String, Map<String, Run>> runs;
089    
090            private final XMLReader reader;
091            private ServerResources resources;
092    
093            /**
094             * 
095             * @param uri
096             * @param params
097             */
098            public Server(URI uri, ConnectionParams params) {
099                    // strip out username and password if present in server URI
100                    this.uri = URIUtils.stripUserInfo(uri);
101    
102                    connection = ConnectionFactory.getConnection(this.uri, params);
103    
104                    reader = new XMLReader(connection);
105                    resources = null;
106    
107                    // initialise run list
108                    runs = new HashMap<String, Map<String, Run>>();
109            }
110    
111            /**
112             * 
113             * @param uri
114             */
115            public Server(URI uri) {
116                    this(uri, null);
117            }

155                    return getRunsFromServer(credentials).values();
156            }
157    
158            /**
159             * Delete all runs on this server instance. Only the runs owned by the
160             * provided credentials will be deleted.
161             * 
162             * @param credentials
163             *            The credentials to authorize the deletion.
164             */
165            public void deleteAllRuns(UserCredentials credentials) {
166                    // Calling getRuns() updates the user's run cache...
167                    for (Run run : getRuns(credentials)) {
168                            run.delete();
169                    }
170    
171                    // ... so we can clear it here now we've deleted them all.
172                    runs.remove(credentials.getUsername());
173            }
174    
175            private Map<String, Run> getRunsFromServer(UserCredentials credentials) {
176                    // Get this user's run list.
177                    URI uri = getLink(ResourceLabel.RUNS);
178                    Map<String, URI> runList = reader.readRunList(uri, credentials);
179    
180                    // Get this user's run cache.
181                    Map<String, Run> userRuns = getUserRunCache(credentials.getUsername());
182    
183                    // Add new runs to the user's run cache.
184                    for (String id : runList.keySet()) {
185                            if (!userRuns.containsKey(id)) {
186                                    userRuns.put(id, new Run(runList.get(id), this, credentials));
187                            }
188                    }
189    
190                    // Any ids in the runs list that aren't in the map we've just got from
191                    // the server are dead and can be removed. Note that we have to use an
192                    // iterator here because we may be removing things as we go.
193                    if (userRuns.size() > runList.size()) {
194                            Iterator<String> iterator = userRuns.keySet().iterator();
195                            while (iterator.hasNext()) {
196                                    String i = iterator.next();
197                                    if (!runList.containsKey(i)) {
198                                            iterator.remove();
199                                    }
200                            }
201                    }
202    
203                    assert (userRuns.size() == runList.size());
204    
205                    return userRuns;
206            }
207    
208            private Map<String, Run> getUserRunCache(String user) {
209                    Map<String, Run> userRuns = runs.get(user);
210                    if (userRuns == null) {
211                            userRuns = new HashMap<String, Run>();
212                            runs.put(user, userRuns);
213                    }
214    
215                    return userRuns;
216            }
217    
218            private ServerResources getServerResources() {
219                    if (resources == null) {
220                            URI restURI = URIUtils.appendToPath(uri, REST_ENDPOINT);
221                            resources = reader.readServerResources(restURI);
222                    }
223    
224                    return resources;
225            }

266             * Get the maximum number of run that this server can host concurrently.
267             * 
268             * @return the maximum number of run that this server can host concurrently.

270            public int getRunLimit(UserCredentials credentials) {
271                    byte[] limit = connection.read(getLink(ResourceLabel.RUNLIMIT),
272                                    MimeType.TEXT, credentials);
273    
274                    return Integer.parseInt(new String(limit).trim());
275            }

277            /**
278             * Initialize a Run on this server instance.
279             * 
280             * @param workflow
281             *            the workflow to be run.
282             * @return the id of the new run as returned by the server.
283             */
284            URI initializeRun(byte[] workflow, UserCredentials credentials) {
285                    URI location = connection.create(getLink(ResourceLabel.RUNS), workflow,
286                                    MimeType.T2FLOW, credentials);
287    
288                    return location;
289            }
290    
291            /**
292             * Create a new Run on this server with the supplied workflow.
293             * 
294             * @param workflow
295             *            the workflow to be run.
296             * @return a new Run instance.
297             */
298            public Run createRun(byte[] workflow, UserCredentials credentials) {
299                    Run run = Run.create(this, workflow, credentials);
300    
301                    getUserRunCache(credentials.getUsername())
302                    .put(run.getIdentifier(), run);
303    
304                    return run;
305            }
306    
307            /**
308             * Create a new Run on this server with the supplied workflow file.
309             * 
310             * @param workflow
311             *            the workflow file to be run.
312             * @return a new Run instance.
313             * @throws IOException
314             *             on IO errors.
315             */
316            public Run createRun(File workflow, UserCredentials credentials)
317                            throws IOException {
318    
319                    return createRun(FileUtils.readFileToByteArray(workflow), credentials);
320            }

358            URI uploadData(URI uri, InputStream stream, String remoteName,
359                            UserCredentials credentials) {
360                    uri = URIUtils.appendToPath(uri, remoteName);
361    
362                    return connection.update(uri, stream, MimeType.BYTES, credentials);
363            }
364    
365            String uploadFile(URI uri, File file, String rename,
366                            UserCredentials credentials) throws FileNotFoundException {
367                    if (file.isDirectory()) {
368                            throw new FileNotFoundException("File passed in is a directory.");
369                    }
370    
371                    if (rename == null || rename.equals("")) {
372                            rename = file.getName();
373                    }
374    
375                    uri = URIUtils.appendToPath(uri, rename);
376    
377                    InputStream is = new FileInputStream(file);
378                    try {
379                            connection.update(uri, is, file.length(), MimeType.BYTES,
380                                            credentials);
381                    } finally {
382                            IOUtils.closeQuietly(is);
383                    }
384    
385                    return rename;
386            }
387    
388            URI mkdir(URI root, String name, UserCredentials credentials) {
389                    if (name.contains("/")) {
390                            throw new IllegalArgumentException(
391                                            "creation of subdirectories directly (" + name + ")");
392                    }
393    
394                    return connection.create(root, XMLWriter.mkdir(name), MimeType.XML,
395                                    credentials);
396            }
397    
398            XMLReader getXMLReader() {
399                    return reader;
400            }

402            private URI getLink(ResourceLabel key) {
403                    return getServerResources().get(key);
404            }
