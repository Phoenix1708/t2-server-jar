033    package uk.org.taverna.server.client;
034    
035    /**
036     * This exception is thrown when a user, who is not the owner of a {@link Run},
037     * tries to access methods that manipulate the security context of a {@link Run}
038     * .
039     * 
040     * Examples of such methods are those which grant or revoke permissions to other
041     * users (e.g. {@link Run#setPermission(String, RunPermission)} or add
042     * credentials to authenticate secure Web Services.
043     * 
044     * @author Robert Haines
045     * @since 0.9.0
046     */
047    public final class IllegalUserAccessException extends ServerException {
048            private static final long serialVersionUID = 1L;
049    
050            private static final String MESSAGE = "User '%s' cannot access that resource.";
051    
052            /**
053             * Construct the exception with the specified username.
054             * 
055             * @param username
056             *            the name of the user that has tried to access a method that
057             *            requires them to be the owner of the Run.
058             */
059            public IllegalUserAccessException(String username) {
060                    super(String.format(MESSAGE, username));
061            }
062    }
