094            public Object read(URI uri, UserCredentials credentials) {
095                    Object resources = null;
096                    InputStream is = connection.readStream(uri, MimeType.XML, credentials);
097    
098                    try {
099                            JAXBContext context = JAXBContext.newInstance(CTX_PATH);
100                            Unmarshaller unmarshaller = context.createUnmarshaller();
101                            resources = unmarshaller.unmarshal(is);
102                    } catch (JAXBException e) {
103                            // TODO Auto-generated catch block
104                            e.printStackTrace();
105                    } finally {
106                            IOUtils.closeQuietly(is);
107                    }
108    
109                    return resources;
110            }
111    
112            public Object read(URI uri) {
113                    return read(uri, null);
114            }

116            public ServerResources readServerResources(URI uri) {
117                    Map<ResourceLabel, URI> links = new HashMap<ResourceLabel, URI>();
118    
119                    // Read server top-level description.
120                    ServerDescription sd = (ServerDescription) read(uri);
121                    String version = sd.getServerVersion();
122                    String revision = sd.getServerRevision();
123                    String timestamp = sd.getServerBuildTimestamp();
124                    links.put(ResourceLabel.RUNS, sd.getRuns().getHref());
125                    links.put(ResourceLabel.POLICY, sd.getPolicy().getHref());
126                    links.put(ResourceLabel.FEED, sd.getFeed().getHref());
127    
128                    // Read policy description and add links to server's set.
129                    PolicyDescription pd = (PolicyDescription) read(links
130                                    .get(ResourceLabel.POLICY));
131                    links.put(ResourceLabel.RUNLIMIT, pd.getRunLimit().getHref());
132                    links.put(ResourceLabel.PERMITTED_WORKFLOWS, pd.getPermittedWorkflows()
133                                    .getHref());
134                    links.put(ResourceLabel.PERMITTED_LISTENERS, pd
135                                    .getPermittedListenerTypes().getHref());
136                    links.put(ResourceLabel.ENABLED_NOTIFICATIONS, pd
137                                    .getEnabledNotificationFabrics().getHref());
138    
139                    return new ServerResources(links, version, revision, timestamp);
140            }

142            public Map<String, URI> readRunList(URI uri, UserCredentials credentials) {
143                    RunList runList = (RunList) read(uri, credentials);
144                    List<TavernaRun> trs = runList.getRun();
145    
146                    Map<String, URI> runs = new HashMap<String, URI>(trs.size());
147    
148                    for (TavernaRun tr : trs) {
149                            runs.put(tr.getValue(), tr.getHref());
150                    }
151    
152                    return runs;
153            }

155            public RunResources readRunResources(URI uri, UserCredentials credentials) {
156                    Map<ResourceLabel, URI> links = new HashMap<ResourceLabel, URI>();
157    
158                    // Read run top-level description.
159                    RunDescription rd = (RunDescription) read(uri, credentials);
160                    String owner = rd.getOwner();
161                    links.put(ResourceLabel.WORKFLOW, rd.getCreationWorkflow().getHref());
162                    links.put(ResourceLabel.CREATE_TIME, rd.getCreateTime().getHref());
163                    links.put(ResourceLabel.START_TIME, rd.getStartTime().getHref());
164                    links.put(ResourceLabel.FINISH_TIME, rd.getFinishTime().getHref());
165                    links.put(ResourceLabel.STATUS, rd.getStatus().getHref());
166                    links.put(ResourceLabel.INPUT, rd.getInputs().getHref());
167                    links.put(ResourceLabel.OUTPUT, rd.getOutput().getHref());
168                    links.put(ResourceLabel.WDIR, rd.getWorkingDirectory().getHref());
169                    links.put(ResourceLabel.EXPIRY, rd.getExpiry().getHref());
170                    links.put(ResourceLabel.SECURITY_CTX, rd.getSecurityContext().getHref());
171    
172                    // Read the inputs description.
173                    JAXBElement<?> root = (JAXBElement<?>) read(
174                                    links.get(ResourceLabel.INPUT), credentials);
175                    TavernaRunInputs tri = (TavernaRunInputs) root.getValue();
176                    links.put(ResourceLabel.BACLAVA, tri.getBaclava().getHref());
177                    links.put(ResourceLabel.EXPECTED_INPUTS, tri.getExpected().getHref());
178    
179                    // Read the special IO listeners - this is kind of hard-coded for now.
180                    for (Location loc : rd.getListeners().getListener()) {
181                            URI u = loc.getHref();
182                            if (URIUtils.extractFinalPathComponent(u).equalsIgnoreCase("io")) {
183                                    root = (JAXBElement<?>) read(u, credentials);
184                                    ListenerDescription ld = (ListenerDescription) root.getValue();
185    
186                                    for (PropertyDescription pd : ld.getProperties().getProperty()) {
187                                            if (pd.getName().equalsIgnoreCase("stdout")) {
188                                                    links.put(ResourceLabel.STDOUT, pd.getHref());
189                                            } else if (pd.getName().equalsIgnoreCase("stderr")) {
190                                                    links.put(ResourceLabel.STDERR, pd.getHref());
191                                            } else if (pd.getName().equalsIgnoreCase("exitcode")) {
192                                                    links.put(ResourceLabel.EXITCODE, pd.getHref());
193                                            }
194                                    }
195                            }
196                    }
197    
198                    // Read the security context iff we are the owner of the run
199                    if (credentials.getUsername().equals(owner)) {
200                            root = (JAXBElement<?>) read(links.get(ResourceLabel.SECURITY_CTX),
201                                            credentials);
202                            SecurityDescriptor sd = (SecurityDescriptor) root.getValue();
203    
204                            links.put(ResourceLabel.PERMISSIONS, sd.getPermissions().getHref());
205                            links.put(ResourceLabel.CREDENTIALS, sd.getCredentials().getHref());
206                            links.put(ResourceLabel.TRUSTS, sd.getTrusts().getHref());
207                    }
208    
209                    return new RunResources(links, owner);
210            }

212            public Map<String, InputPort> readInputPortDescription(Run run, URI uri,
213                            UserCredentials credentials) {
214                    JAXBElement<?> root = (JAXBElement<?>) read(uri, credentials);
215                    InputDescription id = (InputDescription) root.getValue();
216    
217                    Map<String, InputPort> ports = new HashMap<String, InputPort>();
218                    for (uk.org.taverna.server.client.xml.port.InputPort ip : id.getInput()) {
219                            InputPort port = Port
220                                            .newInputPort(run, ip.getName(), ip.getDepth());
221                            ports.put(port.getName(), port);
222                    }
223    
224                    return ports;
225            }
226    
227            public Map<String, OutputPort> readOutputPortDescription(Run run, URI uri,
228                            UserCredentials credentials) {
229                    JAXBElement<?> root = (JAXBElement<?>) read(uri, credentials);
230                    OutputDescription od = (OutputDescription) root.getValue();
231    
232                    Map<String, OutputPort> ports = new HashMap<String, OutputPort>();
233                    for (uk.org.taverna.server.client.xml.port.OutputPort op : od
234                                    .getOutput()) {
235    
236                            LeafValue v = op.getValue();
237                            ListValue lv = op.getList();
238                            ErrorValue ev = op.getError();
239    
240                            PortValue value = null;
241                            if (v != null) {
242                                    value = PortValue.newPortData(run, v.getHref(),
243                                                    v.getContentType(), v.getContentByteLength());
244                            } else if (lv != null) {
245                                    value = parseOutputPortValueStructure(run, lv);
246                            } else if (ev != null) {
247                                    value = PortValue.newPortError(run, ev.getHref(),
248                                                    ev.getErrorByteLength());
249                            }
250    
251                            OutputPort port = Port.newOutputPort(run, op.getName(),
252                                            op.getDepth(), value);
253                            ports.put(port.getName(), port);
254                    }
255    
256                    return ports;
257            }

259            /*
260             * This method has to parse the OutputPort structure by trying to cast to
261             * each type that a port can be. Not pretty.
262             * 
263             * Even though we know that first time through this method we must have a
264             * list we try to cast to a value first as this is what we will most often
265             * have.
266             */
267            private PortValue parseOutputPortValueStructure(Run run, Value value) {
268                    if (LeafValue.class.isInstance(value)) {
269                            LeafValue lv = (LeafValue) value;
270    
271                            return PortValue.newPortData(run, lv.getHref(),
272                                            lv.getContentType(),
273                                            lv.getContentByteLength());
274                    } else if (ListValue.class.isInstance(value)) {
275                            ListValue lv = (ListValue) value;
276    
277                            List<PortValue> list = new ArrayList<PortValue>();
278                            for (Value v : lv.getValueOrListOrError()) {
279                                    list.add(parseOutputPortValueStructure(run, v));
280                            }
281    
282                            return PortValue.newPortList(run, lv.getHref(), list);
283                    } else if (ErrorValue.class.isInstance(value)) {
284                            ErrorValue ev = (ErrorValue) value;
285    
286                            return PortValue.newPortError(run, ev.getHref(),
287                                            ev.getErrorByteLength());
288                    }
289    
290                    // We should NOT get here!
291                    return null;
292            }

294            public Map<String, RunPermission> readRunPermissions(URI uri,
295                            UserCredentials credentials) {
296                    JAXBElement<?> root = (JAXBElement<?>) read(uri, credentials);
297                    PermissionsDescription pd = (PermissionsDescription) root.getValue();
298    
299                    Map<String, RunPermission> perms = new HashMap<String, RunPermission>();
300                    for (LinkedPermissionDescription lpd : pd.getPermission()) {
301                            Permission perm = lpd.getPermission();
302                            perms.put(lpd.getUserName(), RunPermission.fromString(perm.value()));
303                    }
304    
305                    return perms;
306            }

308            public Map<URI, URI> readRunServiceCredentials(URI uri,
309                            UserCredentials credentials) {
310                    JAXBElement<?> root = (JAXBElement<?>) read(uri, credentials);
311                    CredentialList cl = (CredentialList) root.getValue();
312    
313                    Map<URI, URI> creds = new HashMap<URI, URI>();
314                    for (Credential cred : cl.getCredential()) {
315                            CredentialDescriptor cd = cred.getUserpass();
316                            if (cd == null) {
317                                    cd = cred.getKeypair();
318                            }
319    
320                            creds.put(cd.getServiceURI(), cd.getHref());
321                    }
322    
323                    return creds;
324            }

326            public List<URI> readRunTrustedIdentities(URI uri, UserCredentials credentials) {
327                    JAXBElement<?> root = (JAXBElement<?>) read(uri, credentials);
328                    TrustList tl = (TrustList) root.getValue();
329    
330                    List<URI> trusts = new ArrayList<URI>();
331                    for (TrustDescriptor td : tl.getTrust()) {
332                            trusts.add(td.getHref());
333                    }
334    
335                    return trusts;
336            }
