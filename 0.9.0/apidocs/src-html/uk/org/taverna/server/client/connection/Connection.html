033    package uk.org.taverna.server.client.connection;
034    
035    import java.io.InputStream;
036    import java.net.URI;
037    
038    import org.apache.commons.lang.math.LongRange;
039    
040    /**
041     * 
042     * @author Robert Haines
043     */
044    public interface Connection {
045            public InputStream readStream(URI uri, MimeType type, LongRange range,
046                            UserCredentials credentials);
047    
048            public InputStream readStream(URI uri, MimeType type,
049                            UserCredentials credentials);
050    
051            public byte[] read(URI uri, MimeType type, LongRange range,
052                            UserCredentials credentials);
053    
054            public byte[] read(URI uri, MimeType type, UserCredentials credentials);
055    
056            public URI update(URI uri, InputStream content, long length,
057                            MimeType type, UserCredentials credentials);
058    
059            public URI update(URI uri, byte[] content, MimeType type,
060                            UserCredentials credentials);
061    
062            public URI update(URI uri, InputStream content, MimeType type,
063                            UserCredentials credentials);
064    
065            public boolean delete(URI uri, UserCredentials credentials);
066    
067            public URI create(URI uri, InputStream content, long length, MimeType type,
068                            UserCredentials credentials);
069    
070            public URI create(URI uri, byte[] content, MimeType type,
071                            UserCredentials credentials);
072    
073            public URI create(URI uri, InputStream content, MimeType type,
074                            UserCredentials credentials);
075    }
