033    package uk.org.taverna.server.client;
034    
035    /**
036     * This is the abstract superclass of all the exceptions generated by code in
037     * this package.
038     * 
039     * @author Robert Haines
040     */
041    public abstract class ServerException extends RuntimeException {
042            private static final long serialVersionUID = 1L;
043    
044            /**
045             * Constructs a new server exception with the specified detail message.
046             * 
047             * @param message
048             */
049            public ServerException(String message) {
050                    super(message);
051            }
052    }
