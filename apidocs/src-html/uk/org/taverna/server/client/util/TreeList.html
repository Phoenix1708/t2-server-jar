033    package uk.org.taverna.server.client.util;
034    
035    import java.util.ArrayList;
036    import java.util.Arrays;
037    import java.util.Iterator;
038    import java.util.List;
039    
040    /**
041     * The TreeList class represents an arbitrary depth list structure where each
042     * node, including the root, can be either a single value, or a list of child
043     * nodes.
044     * 
045     * The list of child node, should it exist, acts like a {@link java.util.List}.
046     * 
047     * @author Robert Haines
048     */
049    public class TreeList<E> implements Cloneable, Iterable<TreeList<E>> {
050    
051            // the children of this node
052            protected List<TreeList<E>> branch;
053    
054            // the payload of this node
055            protected E leaf;
056    
057            // can this node be both a leaf and a branch?
058            private final boolean composite;
059    
060            /**
061             * Create a new branch node with no initial value.
062             */
063            public TreeList() {
064                    this(false);
065            }
066    
067            /**
068             * Create a new node with no initial value.
069             * 
070             * @param composite
071             *            true for this node to be composite, false otherwise.
072             */
073            public TreeList(boolean composite) {
074                    this.composite = composite;
075                    leaf = null;
076                    branch = new ArrayList<TreeList<E>>();
077            }
078    
079            /**
080             * Create a new leaf node with an initial value.
081             * 
082             * @param value
083             *            the initial value.
084             */
085            public TreeList(E value) {
086                    this(false, value);
087            }
088    
089            /**
090             * Create a new node with an initial value.
091             * 
092             * @param composite
093             *            true for this node to be composite, false otherwise.
094             * @param value
095             *            the initial value.
096             */
097            public TreeList(boolean composite, E value) {
098                    this.composite = composite;
099                    leaf = value;
100                    branch = composite ? new ArrayList<TreeList<E>>() : null;
101            }
102    
103            /**
104             * Create a new node with an initial branch of child nodes.
105             * 
106             * @param composite
107             *            true for this node to be composite, false otherwise.
108             * @param list
109             *            the initial branch of child nodes.
110             */
111            protected TreeList(boolean composite, List<TreeList<E>> list) {
112                    this.composite = composite;
113                    leaf = null;
114                    branch = list != null ? list : new ArrayList<TreeList<E>>();
115            }
116    
117            /**
118             * Create a new node with an initial branch of child nodes.
119             * 
120             * @param list
121             *            the initial branch of child nodes.
122             */
123            protected TreeList(List<TreeList<E>> list) {
124                    this(false, list);
125            }
126    
127            /**
128             * Create a new node with an initial value and branch of child nodes.
129             * 
130             * @param value
131             *            the initial value.
132             * @param list
133             *            the initial branch of child nodes.
134             */
135            protected TreeList(E value, List<TreeList<E>> list) {
136                    composite = true;
137                    leaf = value;
138                    branch = list != null ? list : new ArrayList<TreeList<E>>();
139            }
140    
141            /**
142             * Is this node a leaf? Composite nodes are both leaf and branch nodes.
143             * 
144             * @return true if this node is a leaf, false otherwise.
145             */
146            public boolean isLeaf() {
147                    return composite || branch == null;
148            }
149    
150            /**
151             * Is this node a branch? Composite nodes are both leaf and branch nodes.
152             * 
153             * @return true if this node is a branch, false otherwise.
154             */
155            public boolean isBranch() {
156                    return composite || branch != null;
157            }
158    
159            /**
160             * Is this node a composite node?
161             * 
162             * @return true if this is a composite node, false otherwise.
163             */
164            public boolean isComposite() {
165                    return composite;
166            }
167    
168            /**
169             * Get the value stored in this node.
170             * 
171             * @param coords
172             *            the coordinates of the deep value to retrieve. For a leaf node
173             *            this parameter is not used.
174             * @return the value stored in this node.
175             * @throws TreeListTypeMismatchException
176             *             if the dimensions of the coordinates do not match the node
177             *             type.
178             */
179            public E getValue(int... coords) throws TreeListTypeMismatchException {
180                    // If called without any coords then we are expecting a leaf node.
181                    if (coords.length == 0) {
182                            if (!isLeaf()) {
183                                    throw new TreeListTypeMismatchException("leaf");
184                            }
185    
186                            return leaf;
187                    }
188    
189                    // Otherwise we are expecting a branch node.
190                    if (!isBranch()) {
191                            throw new TreeListTypeMismatchException("branch");
192                    }
193    
194                    int[] newCoords = Arrays.copyOfRange(coords, 1, coords.length);
195                    return getNode(coords[0]).getValue(newCoords);
196            }
197    
198            /**
199             * Set the value stored in this node.
200             * 
201             * @param data
202             *            the new value of this node.
203             * @return the old value of this node.
204             * @throws TreeListTypeMismatchException
205             *             if this is not a leaf node.
206             */
207            public E setValue(E data) throws TreeListTypeMismatchException {
208                    if (!isLeaf()) {
209                            throw new TreeListTypeMismatchException("leaf");
210                    }
211    
212                    E temp = leaf;
213                    leaf = data;
214                    return temp;
215            }
216    
217            /**
218             * Get a node from the child branch of this node.
219             * 
220             * @param index
221             *            the index of the node in the child branch to return.
222             * @return the node from the child branch of this node.
223             * @throws TreeListTypeMismatchException
224             *             if this is not a branch node.
225             */
226            public TreeList<E> getNode(int index) throws TreeListTypeMismatchException {
227                    if (!isBranch()) {
228                            throw new TreeListTypeMismatchException("branch");
229                    }
230    
231                    return branch.get(index);
232            }
233    
234            /**
235             * Set a node in the child branch of this node.
236             * 
237             * @param index
238             *            the index of the node in the child branch to change.
239             * @param node
240             *            the new node.
241             * @return the old node.
242             * @throws TreeListTypeMismatchException
243             *             if this is not a branch node.
244             */
245            public TreeList<E> setNode(int index, TreeList<E> node)
246                            throws TreeListTypeMismatchException {
247                    if (!isBranch()) {
248                            throw new TreeListTypeMismatchException("branch");
249                    }
250    
251                    TreeList<E> temp = branch.get(index);
252                    branch.set(index, node);
253                    return temp;
254            }
255    
256            /**
257             * Add a node to the end of this node's child branch.
258             * 
259             * @param node
260             *            the node to add.
261             * @return true
262             * @throws TreeListTypeMismatchException
263             *             if this is not a branch node.
264             */
265            public boolean addNode(TreeList<E> node)
266                            throws TreeListTypeMismatchException {
267                    if (!isBranch()) {
268                            throw new TreeListTypeMismatchException("branch");
269                    }
270    
271                    return branch.add(node);
272            }
273    
274            /**
275             * Insert a node at the specified position in the child branch. Shifts the
276             * node currently at that position (if any) and any subsequent nodes to the
277             * right (adds one to their indices).
278             * 
279             * @param index
280             *            index at which the specified node is to be inserted.
281             * @param node
282             *            node to be inserted.
283             * @throws TreeListTypeMismatchException
284             *             if this is not a branch node.
285             */
286            public void addNode(int index, TreeList<E> node)
287                            throws TreeListTypeMismatchException {
288                    if (!isBranch()) {
289                            throw new TreeListTypeMismatchException("branch");
290                    }
291    
292                    branch.add(index, node);
293            }
294    
295            /**
296             * Removes the node at the specified position in this list. Shifts any
297             * subsequent nodes to the left (subtracts one from their indices). Returns
298             * the node that was removed from the list.
299             * 
300             * @param index
301             *            the index of the node to be removed.
302             * @return the node previously at the specified position.
303             * @throws TreeListTypeMismatchException
304             *             if this is not a branch node.
305             */
306            public TreeList<E> removeNode(int index)
307                            throws TreeListTypeMismatchException {
308                    if (!isBranch()) {
309                            throw new TreeListTypeMismatchException("branch");
310                    }
311    
312                    return branch.remove(index);
313            }
314    
315            /**
316             * Removes the first occurrence of the specified node from this list, if it
317             * is present (optional operation). If this list does not contain the node,
318             * it is unchanged.
319             * 
320             * @param o
321             *            node to be removed from this list, if present.
322             * @return true if this list contained the specified element.
323             * @throws TreeListTypeMismatchException
324             *             if this is not a branch node.
325             */
326            public boolean removeNode(Object o) throws TreeListTypeMismatchException {
327                    if (!isBranch()) {
328                            throw new TreeListTypeMismatchException("branch");
329                    }
330    
331                    return branch.remove(o);
332            }
333    
334            /**
335             * Returns true if this list contains the specified node. More formally,
336             * returns true if and only if the child branch of this node contains at
337             * least one node e such that (o==null ? e==null : o.equals(e)).
338             * 
339             * @param o
340             *            node whose presence in this list is to be tested.
341             * @return true if this list contains the specified element.
342             * @throws TreeListTypeMismatchException
343             *             if this is not a branch node.
344             */
345            public boolean containsNode(Object o) throws TreeListTypeMismatchException {
346                    if (!isBranch()) {
347                            throw new TreeListTypeMismatchException("branch");
348                    }
349    
350                    return branch.contains(o);
351            }
352    
353            /**
354             * Removes all of the nodes from this node's branch and sets the value of
355             * this node to null.
356             */
357            public void clear() {
358                    leaf = null;
359    
360                    if (isBranch()) {
361                            branch.clear();
362                    }
363            }
364    
365            /**
366             * Returns the number of nodes on this branch or 0 if it is a leaf nodes.
367             * 
368             * @return the number of nodes on this branch or 0 if it is a leaf nodes.
369             */
370            public int size() {
371                    return isBranch() ? branch.size() : 0;
372            }
373    
374            /**
375             * Returns a string representation of this TreeList. If this node is a leaf
376             * it simply returns a representation of its value. If it is a branch then
377             * it recursively calls toString() on each child.
378             * 
379             * @return a string representation of this TreeList.
380             */
381            @Override
382            public String toString() {
383                    String result = "";
384    
385                    if (isLeaf()) {
386                            result = leaf == null ? "null" : leaf.toString();
387                    }
388    
389                    if (isComposite()) {
390                            result += ": ";
391                    }
392    
393                    if (isBranch()) {
394                            if (branch.size() == 0) {
395                                    result += "[null]";
396                            } else {
397                                    result += "[" + branch.get(0);
398                                    for (int i = 1; i < branch.size(); i++) {
399                                            result += ", " + branch.get(i);
400                                    }
401                                    result += "]";
402                            }
403                    }
404    
405                    return result;
406            }
407    
408            /**
409             * Returns a shallow copy of this TreeList instance. (The value and node
410             * elements themselves are not copied.)
411             * 
412             * @return a clone of this TreeList instance
413             */
414            @Override
415            public Object clone() {
416                    if (this.isComposite()) {
417                            return new TreeList<E>(this.leaf, this.branch);
418                    } else {
419                            if (this.isLeaf()) {
420                                    return new TreeList<E>(this.leaf);
421                            } else {
422                                    return new TreeList<E>(this.branch);
423                            }
424                    }
425            }
426    
427            /**
428             * Returns an iterator over the child nodes of this node, if it has any.
429             * 
430             * @return an iterator.
431             */
432            @Override
433            public Iterator<TreeList<E>> iterator()
434                            throws TreeListTypeMismatchException {
435                    if (!isBranch()) {
436                            throw new TreeListTypeMismatchException("branch");
437                    }
438    
439                    return branch.iterator();
440            }
441    }
