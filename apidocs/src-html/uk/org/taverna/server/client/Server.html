057    /**
058     * The Server class represents a connection to a Taverna Server instance
059     * somewhere on the Internet. Only one instance of this class is created for
060     * each Taverna Server instance.
061     * 
062     * To make a connection to a server call {@link Server#connect(URI)} with it
063     * full URL as the parameter. If there already exists a Server instance that is
064     * connected to this Taverna Server then it will be returned, otherwise a new
065     * Server instance is created and returned.
066     * 
067     * @author Robert Haines
068     */
069    public final class Server {
070            private final Connection connection;
071

073            private final float version;
074            private final Map<String, Run> runs;
075    
076            private final Map<String, String> links;

080            /**
081             * 
082             * @param uri
083             * @param params
084             */
085            public Server(URI uri, ConnectionParams params) {
086                    // strip out username and password if present in server URI
087                    String userInfo = uri.getUserInfo();
088                    if (userInfo != null) {
089                            try {
090                                    this.uri = new URI(uri.getScheme(), null, uri.getHost(),
091                                                    uri.getPort(), uri.getPath(), null, null);
092                            } catch (URISyntaxException e) {
093                                    throw new IllegalArgumentException("Bad URI passed in: " + uri);
094                            }
095                    } else {
096                            this.uri = uri;
097                    }
098    
099                    connection = ConnectionFactory.getConnection(this.uri, params);
100                    xmlUtils = XmlUtils.getInstance();
101    
102                    // add a slash to the end of this address to work around this bug:
103                    // http://www.mygrid.org.uk/dev/issues/browse/TAVSERV-113
104                    String restPath = this.uri.toASCIIString() + "/rest/";
105                    Document doc = ParseUtil.parse(new String(connection.getAttribute(
106                                    restPath, null)));
107                    version = getServerVersion(doc);
108                    links = getServerDescription(doc);
109    
110                    // System.out.println("u: " + this.uri);
111                    // System.out.println("v: " + version);
112                    // for (String s : links.keySet()) {
113                    // System.out.println(s + ": " + links.get(s));
114                    // }
115    
116                    // initialise run list
117                    runs = new HashMap<String, Run>();
118                    // getRunsFromServer();
119            }

159             * Delete a Run from the server.
160             * 
161             * @param id
162             *            The id of the run to delete.
163             */
164            public void deleteRun(String id, UserCredentials credentials) {
165                    try {
166                            connection.delete(links.get("runs") + "/" + id, credentials);
167                    } catch (AccessForbiddenException e) {
168                            if (getRunsFromServer(credentials).containsKey(id)) {
169                                    throw e;
170                            } else {
171                                    throw new RunNotFoundException(id);
172                            }
173                    }
174            }

195            private Map<String, Run> getRunsFromServer(UserCredentials credentials) {
196                    String runList = new String(connection.getAttribute(links.get("runs"),
197                                    credentials));
198                    Document doc = ParseUtil.parse(runList);
199    
200                    // add new runs, but keep a list of the new
201                    // ids so we can remove the stale ones below.
202                    String id;
203                    ArrayList<String> ids = new ArrayList<String>();
204                    for (Element e : xmlUtils.evalXPath(doc, "//nsr:run")) {
205                            id = e.getTextContent().trim();
206                            ids.add(id);
207                            if (!runs.containsKey(id)) {
208                                    runs.put(id, new Run(this, credentials, id));
209                            }
210                    }
211    
212                    // any ids in the runs list that aren't in the list we've
213                    // just got from the server are dead and can be removed.
214                    if (runs.size() > ids.size()) {
215                            for (String i : runs.keySet()) {
216                                    if (!ids.contains(i)) {
217                                            runs.remove(i);
218                                    }
219                            }
220                    }
221    
222                    assert (runs.size() == ids.size());
223    
224                    return runs;
225            }

227            private float getServerVersion(Document doc) {
228                    String version = xmlUtils.evalXPath(doc, "/nsr:serverDescription",
229                                    "nss:serverVersion");
230    
231                    if (version.equalsIgnoreCase("")) {
232                            return 1.0f;
233                    } else {
234                            return Float.parseFloat(version.substring(0, 3));
235                    }
236            }
237    
238            private Map<String, String> getServerDescription(Document doc) {
239                    HashMap<String, String> links = new HashMap<String, String>();
240    
241                    links.put("runs", xmlUtils.evalXPath(doc, "//nsr:runs", "xlink:href"));
242    
243                    if (version > 1.0) {
244                            String policy = xmlUtils.evalXPath(doc, "//nsr:policy",
245                                            "xlink:href");
246    
247                            links.put("policy", policy);
248                            doc = ParseUtil.parse(new String(connection.getAttribute(policy,
249                                            null)));
250    
251                            links.put("permlisteners", xmlUtils.evalXPath(doc,
252                                            "//nsr:permittedListenerTypes", "xlink:href"));
253    
254                            links.put("notifications", xmlUtils.evalXPath(doc,
255                                            "//nsr:enabledNotificationFabrics", "xlink:href"));
256                    } else {
257                            links.put("permlisteners", xmlUtils.evalXPath(doc,
258                                            "//nsr:permittedListeners", "xlink:href"));
259                    }
260    
261                    links.put("runlimit",
262                                    xmlUtils.evalXPath(doc, "//nsr:runLimit", "xlink:href"));
263                    links.put("permworkflows", xmlUtils.evalXPath(doc,
264                                    "//nsr:permittedWorkflows", "xlink:href"));
265    
266                    return links;
267            }

326             * 
327             * @param workflow
328             *            the workflow to be run.
329             * @return the id of the new run as returned by the server.
330             */
331            String initializeRun(String workflow, UserCredentials credentials) {
332                    String id = null;
333                    String location = connection.upload(links.get("runs"),
334                                    xmlUtils.buildXMLFragment("workflow", workflow), credentials);
335    
336                    if (location != null) {
337                            id = location.substring(location.lastIndexOf("/") + 1);
338                    }
339    
340                    return id;
341            }

526            /**
527             * Upload a file to the server for use by a run.
528             * 
529             * @param id
530             *            the id of the run to upload to.
531             * @param file
532             *            the file to upload.
533             * @param uploadLocation
534             *            the location to upload to. This should be the full URI.
535             * @param rename
536             *            optionally rename the file at the remote location. Pass null
537             *            or the empty string to ignore.
538             * @return the name of the file on the remote server. This will be unchanged
539             *         unless rename was used.
540             * @throws IOException
541             * @see #uploadRunFile(id, File, String)
542             * @see #uploadRunFile(Run, File, String, String)
543             * @see #uploadRunFile(Run, File, String)
544             */
545            public String uploadRunFile(String id, File file, String uploadLocation,
546                            String rename, UserCredentials credentials) throws IOException {
547    
548                    if (rename == null || rename.equals("")) {
549                            rename = file.getName();
550                    }
551    
552                    byte[] data = FileUtils.readFileToByteArray(file);
553                    String contents = Base64.encodeBase64String(data);
554    
555                    connection.upload(uploadLocation,
556                                    xmlUtils.buildXMLFragment("upload", rename, contents),
557                                    credentials);
558    
559                    return rename;
560            }

630            void makeRunDir(String id, String root, String name,
631                            UserCredentials credentials) throws IOException {
632                    if (name.contains("/")) {
633                            throw new AccessForbiddenException(
634                                            "creation of subdirectories directly (" + name + ")");
635                    }
636    
637                    try {
638                            connection.upload(root, xmlUtils.buildXMLFragment("mkdir", name),
639                                            credentials);
640                    } catch (AttributeNotFoundException e) {
641                            if (getRunsFromServer(credentials).containsKey(id)) {
642                                    throw e;
643                            } else {
644                                    throw new RunNotFoundException(id);
645                            }
646                    }
647            }
648    
649            void makeRunDir(Run run, String root, String name,
650                            UserCredentials credentials) throws IOException {
651                    makeRunDir(run.getIdentifier(), root, name, credentials);
652            }
