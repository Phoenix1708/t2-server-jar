001    package uk.org.taverna.server.client;
002    
003    import java.net.URI;
004    import java.util.List;
005    
006    import net.sf.practicalxml.util.NodeListIterable;
007    
008    import org.w3c.dom.Element;
009    import org.w3c.dom.Node;
010    
011    import uk.org.taverna.server.client.util.TreeList;
012    
013    public final class OutputPort extends Port {
014    
015            private final TreeList<PortValue> structure;
016            private int totalDataSize;
017            private boolean error;
018    
019            /**
020             * 
021             * @param run
022             * @param xml
023             */
024            public OutputPort(Run run, Element xml) {
025                    super(run, xml);
026    
027                    totalDataSize = 0;
028                    error = false;
029                    structure = parse((Element) xml.getFirstChild());
030            }
031    
032            /**
033             * 
034             * @param coords
035             * @return
036             * @throws IndexOutOfBoundsException
037             */
038            public PortValue getValue(int... coords) throws IndexOutOfBoundsException {
039                    if (coords.length != depth) {
040                            throw new IndexOutOfBoundsException("OutputPort of depth " + depth
041                                            + " accessed as depth " + coords.length + ".");
042                    }
043    
044                    return structure.getValue(coords);
045            }
046    
047            /**
048             * 
049             * @param coords
050             * @return
051             * @throws IndexOutOfBoundsException
052             */
053            public PortValue getValue(List<Integer> coords)
054                            throws IndexOutOfBoundsException {
055                    if (coords == null || coords.size() == 0) {
056                            return getValue();
057                    } else {
058                            int[] array = new int[coords.size()];
059                            for (int i = 0; i < array.length; i++) {
060                                    array[i] = coords.get(i);
061                            }
062                            return getValue(array);
063                    }
064            }
065    
066            /**
067             * Does this port contain error? should this be hasError?
068             * 
069             * @return <code>true</code> if this port contains an error,
070             *         <code>false</code> otherwise.
071             */
072            public boolean isError() {
073                    return error;
074            }
075    
076            /**
077             * Get the total size of all the data on this OutputPort. For a singleton
078             * port this is simply the size of the single value but for any other depth
079             * port it is the addition of all values in the port.
080             * 
081             * @return The total data size of this OutputPort.
082             */
083            public int getTotalDataSize() {
084                    return totalDataSize;
085            }
086    
087            @Override
088            public String toString() {
089                    return structure.toString();
090            }
091    
092            /*
093             * Side-effect warning: This method sets this.error and this.totalDataSize!
094             */
095            private TreeList<PortValue> parse(Element node) {
096                    String name = node.getNodeName();
097    
098                    if (name.equalsIgnoreCase("port:value")) {
099                            URI ref = URI.create(xmlUtils.getXlinkAttribute(node, "href"));
100                            String type = xmlUtils.getPortAttribute(node, "contentType");
101                            int size = Integer.parseInt(xmlUtils.getPortAttribute(node,
102                                            "contentByteLength"));
103                            totalDataSize += size;
104                            PortValue value = new PortValue(this, ref, type, size);
105                            return new TreeList<PortValue>(value);
106                    } else if (name.equalsIgnoreCase("port:list")) {
107                            TreeList<PortValue> list = new TreeList<PortValue>();
108                            NodeListIterable nodes = new NodeListIterable(node.getChildNodes());
109                            for (Node n : nodes) {
110                                    list.addNode(parse((Element) n));
111                            }
112                            return list;
113                    } else {
114                            this.error = true;
115                            URI error = URI.create(xmlUtils.getXlinkAttribute(node, "href"));
116                            PortValue value = new PortValue(this, error);
117                            return new TreeList<PortValue>(value);
118                    }
119            }
120    }
