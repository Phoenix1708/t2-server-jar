052    /**
053     * The Run class represents a workflow run on a Taverna Server instance. It is

079            /**
080             * Create a new Run instance on the specified server with the supplied
081             * workflow.
082             * 
083             * @param server
084             *            The server to create the Run on.
085             * @param workflow
086             *            The workflow associated with the Run.
087             * @param credentials
088             */
089            public Run(Server server, String workflow, UserCredentials credentials) {
090                    this.server = server;
091                    this.id = server.initializeRun(workflow, credentials);
092                    this.workflow = workflow;
093                    this.baclavaIn = false;
094                    this.baclavaOut = false;
095    
096                    xmlUtils = XmlUtils.getInstance();
097    
098                    this.credentials = credentials;
099                    links = getRunDescription(this.credentials);
100            }

191            /**
192             * Upload a file to this Run instance's workspace on the server.
193             * 
194             * @param file
195             *            The file to upload.
196             * @param remoteDirectory
197             *            The directory within the workspace to upload the file to.
198             * @param rename

203            public String uploadFile(File file, String remoteDirectory, String rename)
204                            throws IOException {
205                    String uploadLocation = links.get("wdir");

223            /**
224             * Upload a baclava file to specify all input port values.
225             * 
226             * @param file
227             *            The file to upload.
228             * @throws IOException
229             */
230            public void setBaclavaInput(File file) throws IOException {
231                    RunStatus rs = getStatus();
232                    if (rs == RunStatus.INITIALIZED) {
233                            String filename = uploadFile(file);
234                            server.setRunAttribute(this, links.get("baclava"), filename,
235                                            "text/plain", credentials);
236    
237                            baclavaIn = true;
238                    } else {
239                            throw new RunStateException(rs, RunStatus.INITIALIZED);
240                    }
241            }

251            /**
252             * Set the server to return outputs for this Run in baclava format. This
253             * must be set before the Run is started.
254             */
255            public void requestBaclavaOutput() {
256                    // don't try and request it again!
257                    if (baclavaOut) {
258                            return;
259                    }
260    
261                    RunStatus rs = getStatus();
262                    if (rs == RunStatus.INITIALIZED) {
263                            server.setRunAttribute(this, links.get("output"), BACLAVA_FILE,
264                                            "text/plain", credentials);
265    
266                            baclavaOut = true;
267                    } else {
268                            throw new RunStateException(rs, RunStatus.INITIALIZED);
269                    }
270            }

272            /**
273             * Get the outputs of this Run as a baclava formatted document. The Run must
274             * have been set to output in baclava format before it is started.
275             * 
276             * @return The baclava formatted document contents as a String.
277             * @see #setBaclavaOutput()
278             * @see #setBaclavaOutput(String)
279             */
280            public String getBaclavaOutput() {
281                    RunStatus rs = getStatus();
282                    if (rs == RunStatus.FINISHED) {
283                            String baclavaLink = links.get("wdir") + "/" + BACLAVA_FILE;
284                            if (!baclavaOut) {
285                                    throw new AttributeNotFoundException(baclavaLink);
286                            }
287    
288                            return server.getRunAttribute(this, baclavaLink,
289                                            "application/octet-stream", credentials);
290                    } else {
291                            throw new RunStateException(rs, RunStatus.FINISHED);
292                    }
293            }

341            /**
342             * Start this Run running on the server. The Run must not be already
343             * running, or finished.
344             * 
345             * @throws IOException
346             */
347            public void start() throws IOException {
348                    RunStatus rs = getStatus();
349                    if (rs != RunStatus.INITIALIZED) {
350                            throw new RunStateException(rs, RunStatus.INITIALIZED);
351                    }
352    
353                    // set all the inputs
354                    if (!isBaclavaInput()) {
355                            setAllInputs();
356                    }
357    
358                    server.setRunAttribute(this, links.get("status"),
359                                    RunStatus.RUNNING.status(), "text/plain", credentials);
360            }

466            /**
467             * Create a directory in the workspace of this Run.
468             * 
469             * @param dir
470             *            the name of the directory to create.
471             * @throws IOException
472             */
473            public void mkdir(String dir) throws IOException {
474                    if (dir.contains("/")) {
475                            int lastSlash = dir.lastIndexOf("/");
476                            String leaf = dir.substring(lastSlash + 1, dir.length());
477                            String path = dir.substring(0, lastSlash);
478                            server.makeRunDir(this, links.get("wdir") + "/" + path, leaf,
479                                            credentials);
480                    } else {
481                            server.makeRunDir(this, links.get("wdir"), dir, credentials);
482                    }
483            }

485            /*
486             * Set all the inputs on the server. The inputs must have been set prior to
487             * this call using the InputPort API.
488             */
489            private void setAllInputs() throws IOException {
490                    for (InputPort port : getInputPorts().values()) {
491                            if (!port.isSet()) {
492                                    continue;
493                            }
494    
495                            if (port.isFile()) {
496                                    // If we're using a local file upload it first then set the
497                                    // port to use a remote file.
498                                    if (!port.isRemoteFile()) {
499                                            File file = new File(uploadFile(port.getFile()));
500                                            port.setRemoteFile(file);
501                                    }
502    
503                                    setInputPort(port);
504                            } else {
505                                    setInputPort(port);
506                            }
507                    }
508            }
509    
510            private void setInputPort(InputPort port) {
511                    String path = links.get("inputs") + "/input/" + port.getName();
512                    String value;
513    
514                    if (port.isFile()) {
515                            String payload = xmlUtils.escapeXML(port.getFile().getPath());
516                            value = xmlUtils.buildXMLFragment("inputfile", payload);
517                    } else {
518                            String payload = xmlUtils.escapeXML(port.getValue());
519                            value = xmlUtils.buildXMLFragment("inputvalue", payload);
520                    }
521    
522                    server.setRunAttribute(this, path, value, "application/xml",
523                                    credentials);
524            }

526            private Map<String, String> getRunDescription(UserCredentials credentials) {
527                    HashMap<String, String> links = new HashMap<String, String>();
528    
529                    // parse out the simple stuff
530                    String description = server.getRunDescription(this, credentials);
531                    Document doc = ParseUtil.parse(description);
532    
533                    links.put("expiry",
534                                    xmlUtils.evalXPath(doc, "//nsr:expiry", "xlink:href"));
535                    links.put("workflow",
536                                    xmlUtils.evalXPath(doc, "//nsr:creationWorkflow", "xlink:href"));
537                    links.put("status",
538                                    xmlUtils.evalXPath(doc, "//nsr:status", "xlink:href"));
539                    links.put("createtime",
540                                    xmlUtils.evalXPath(doc, "//nsr:createTime", "xlink:href"));
541                    links.put("starttime",
542                                    xmlUtils.evalXPath(doc, "//nsr:startTime", "xlink:href"));
543                    links.put("finishtime",
544                                    xmlUtils.evalXPath(doc, "//nsr:finishTime", "xlink:href"));
545                    links.put("wdir",
546                                    xmlUtils.evalXPath(doc, "//nsr:workingDirectory", "xlink:href"));
547                    links.put("inputs",
548                                    xmlUtils.evalXPath(doc, "//nsr:inputs", "xlink:href"));
549                    links.put("output",
550                                    xmlUtils.evalXPath(doc, "//nsr:output", "xlink:href"));
551                    links.put("securectx",
552                                    xmlUtils.evalXPath(doc, "//nsr:securityContext", "xlink:href"));
553                    links.put("listeners",
554                                    xmlUtils.evalXPath(doc, "//nsr:listeners", "xlink:href"));
555    
556                    // get the inputs
557                    String inputs = server.getRunAttribute(this, links.get("inputs"),
558                                    "application/xml", credentials);
559                    doc = ParseUtil.parse(inputs);
560                    links.put("baclava",
561                                    xmlUtils.evalXPath(doc, "//nsr:baclava", "xlink:href"));
562                    links.put("inputexp",
563                                    xmlUtils.evalXPath(doc, "//nsr:expected", "xlink:href"));
564    
565                    // set io properties
566                    links.put("io", links.get("listeners") + "/io");
567                    links.put("stdout", links.get("io") + "/properties/stdout");
568                    links.put("stderr", links.get("io") + "/properties/stderr");
569                    links.put("exitcode", links.get("io") + "/properties/exitcode");
570    
571                    return links;
572            }

574            private Map<String, InputPort> getInputPortInfo() {
575                    Map<String, InputPort> ports = new HashMap<String, InputPort>();
576    
577                    String portDesc = server.getRunAttribute(this, links.get("inputexp"),
578                                    "application/xml", credentials);
579                    Document doc = ParseUtil.parse(portDesc);
580    
581                    for (Element e : xmlUtils.evalXPath(doc, "//port:input")) {
582                            InputPort port = new InputPort(this, e);
583                            ports.put(port.getName(), port);
584                    }
585    
586                    return ports;
587            }
588    
589            private Map<String, OutputPort> getOutputPortInfo() {
590                    Map<String, OutputPort> ports = new HashMap<String, OutputPort>();
591    
592                    String portDesc = server.getRunAttribute(this, links.get("output"),
593                                    "application/xml", credentials);
594                    Document doc = ParseUtil.parse(portDesc);
595    
596                    for (Element e : xmlUtils.evalXPath(doc, "//port:output")) {
597                            OutputPort port = new OutputPort(this, e);
598                            ports.put(port.getName(), port);
599                    }
600    
601                    return ports;
602            }
